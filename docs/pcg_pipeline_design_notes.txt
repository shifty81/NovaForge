import json, random, subprocess, os

UNIVERSE_SEED = 987654
random.seed(UNIVERSE_SEED)

galaxy = {"systems": []}

for sys_idx in range(5):  # number of systems
    system_seed = random.randint(0, 999999)
    rng = random.Random(system_seed)
    
    system_id = f"system_{sys_idx}"
    
    # Generate stars
    stars = [{"star_id": f"{system_id}_star_0", "type": rng.choice(["G","K","M"]), "seed": rng.randint(0,999999)}]
    
    # Generate planets
    planets = []
    for i in range(rng.randint(1, 8)):
        planet_seed = rng.randint(0, 999999)
        planets.append({
            "planet_id": f"{system_id}_planet_{i}",
            "type": rng.choice(["terrestrial","gas_giant","ice"]),
            "seed": planet_seed,
            "stations": []
        })
    
    # Generate stations per planet
    for planet in planets:
        station_count = rng.randint(0, 3)
        for j in range(station_count):
            station_seed = rng.randint(0, 999999)
            planet["stations"].append({
                "station_id": f"{planet['planet_id']}_station_{j}",
                "type": rng.choice(["trade","military","refinery"]),
                "seed": station_seed
            })
    
    # Generate ships per system
    ships = []
    for k in range(rng.randint(2, 5)):
        ship_seed = rng.randint(0, 999999)
        ship_class = rng.choice(["Corvette","Frigate","Cruiser","Battleship"])
        ships.append({
            "ship_id": f"{system_id}_ship_{k}",
            "class": ship_class,
            "seed": ship_seed
        })
        # Generate ships via Blender headless
        ship_json = {
            "ship_id": f"{system_id}_ship_{k}",
            "class": ship_class,
            "seed": ship_seed,
            "modules": {"engine":"ion_thruster_mk1", "weapon":["laser_mk1"]},
            "hardpoints": rng.randint(2, 6)
        }
        os.makedirs("build/ships", exist_ok=True)
        ship_json_path = f"build/ships/{ship_json['ship_id']}.json"
        with open(ship_json_path, "w") as f:
            json.dump(ship_json, f)
        subprocess.run([
            "blender", "--background", "--python", "generate_ship.py", "--",
            "--ship_json", ship_json_path,
            "--output_dir", "build/ships"
        ])
    
    galaxy["systems"].append({
        "system_id": system_id,
        "seed": system_seed,
        "stars": stars,
        "planets": planets,
        "ships": ships
    })

# Save universe data
with open("build/universe.json", "w") as f:
    json.dump({"universe_seed": UNIVERSE_SEED, "galaxy": galaxy}, f, indent=2)

import bpy, os, json, numpy as np

# -------------------------------
# 1. Generate Planet Mesh from Heightmap
# -------------------------------
def create_planet_mesh(planet_id, heightmap, radius=1.0):
    size = heightmap.shape[0]
    mesh_name = f"{planet_id}_mesh"
    mesh = bpy.data.meshes.new(mesh_name)
    obj = bpy.data.objects.new(mesh_name, mesh)
    bpy.context.collection.objects.link(obj)
    
    verts = []
    faces = []

    # Simple grid-to-sphere projection
    for y in range(size):
        for x in range(size):
            h = heightmap[y, x] * 0.1  # scale height
            theta = (x / size) * 3.14159 * 2
            phi = (y / size) * 3.14159
            vx = (radius + h) * np.sin(phi) * np.cos(theta)
            vy = (radius + h) * np.cos(phi)
            vz = (radius + h) * np.sin(phi) * np.sin(theta)
            verts.append((vx, vy, vz))
    
    for y in range(size-1):
        for x in range(size-1):
            idx = y*size + x
            faces.append((idx, idx+1, idx+size+1, idx+size))
    
    mesh.from_pydata(verts, [], faces)
    mesh.update()
    return obj

# -------------------------------
# 2. Generate Foliage
# -------------------------------
def generate_foliage(obj, foliage_list):
    for f in foliage_list:
        # Create placeholder mesh for tree/rock
        bpy.ops.mesh.primitive_cone_add(vertices=6, radius1=0.01, depth=0.05)
        tree = bpy.context.object
        tree.location = f['position']
        tree.parent = obj

# -------------------------------
def animate_planet(obj, rotation_speed=0.01, orbit_speed=0.001, orbit_radius=1.0):
    # Rotation animation
    obj.rotation_euler[2] = 0
    obj.keyframe_insert(data_path="rotation_euler", frame=0)
    obj.rotation_euler[2] += rotation_speed
    obj.keyframe_insert(data_path="rotation_euler", frame=250)
    
    # Orbit animation along Z axis
    obj.location[0] = orbit_radius
    obj.keyframe_insert(data_path="location", frame=0)
    obj.location[0] = -orbit_radius
    obj.keyframe_insert(data_path="location", frame=250)

# -------------------------------
# 4. Generate LODs
# -------------------------------
def generate_lods(obj, output_dir, lod_count=3):
    lod_files = []
    for i in range(1, lod_count+1):
        ratio = 1.0 / (2**i)
        dec_obj = obj.copy()
        dec_obj.data = obj.data.copy()
        bpy.context.collection.objects.link(dec_obj)
        mod = dec_obj.modifiers.new(name="Decimate", type='DECIMATE')
        mod.ratio = ratio
        bpy.context.view_layer.objects.active = dec_obj
        bpy.ops.object.modifier_apply(modifier=mod.name)
        lod_file = os.path.join(output_dir, f"{obj.name}_LOD{i}.obj")
        bpy.ops.export_scene.obj(filepath=lod_file, use_selection=True)
        lod_files.append(lod_file)
        bpy.data.objects.remove(dec_obj)
    return lod_files

def export_planet(planet_data, output_dir):
