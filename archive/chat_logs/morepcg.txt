#include "NameGenerator.h"

static uint64_t SplitMix64(uint64_t& x) {
    uint64_t z = (x += 0x9e3779b97f4a7c15);
    z = (z ^ (z >> 30)) * 0xbf58476d1ce4e5b9;
    z = (z ^ (z >> 27)) * 0x94d049bb133111eb;
    return z ^ (z >> 31);
}

uint64_t NameGenerator::Hash(uint64_t a, uint64_t b) {
    uint64_t h = a ^ (b + 0x9e3779b97f4a7c15);
    return SplitMix64(h);
}

std::string NameGenerator::Generate(
    uint64_t seed,
    NameStyle style,
    uint32_t minSyllables,
    uint32_t maxSyllables
) {
    PhonemeSet p = GetPhonemes(style);
    uint64_t rng = seed;

    uint32_t syllables = minSyllables +
        (SplitMix64(rng) % (maxSyllables - minSyllables + 1));

    std::string name;
    for (uint32_t i = 0; i < syllables; ++i) {
        name += p.prefixes[SplitMix64(rng) % p.prefixes.size()];
        name += p.cores[SplitMix64(rng) % p.cores.size()];
        name += p.suffixes[SplitMix64(rng) % p.suffixes.size()];
    }

    name[0] = toupper(name[0]);
    return name;
}
